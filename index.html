const { app, BrowserWindow } = require('electron'); const path = require('path'); function createWindow() { const win = new BrowserWindow({ width: 480, height: 700, minWidth: 400, minHeight: 500, backgroundColor: '#0A192F', icon: path.join(__dirname, 'icon.ico'), webPreferences: { nodeIntegration: true, // ВКЛЮЧИТЬ для доступа к require() contextIsolation: false, // ОТКЛЮЧИТЬ для работы с node enableRemoteModule: true // ВКЛЮЧИТЬ }, show: true }); // Загружаем index.html win.loadFile('index.html'); // Убираем стандартное меню win.setMenu(null); // Открываем DevTools для отладки (можно закомментировать позже) win.webContents.openDevTools(); console.log('Window created successfully'); } // Запускаем приложение app.whenReady().then(() => { console.log('Electron app ready'); createWindow(); }); // Закрываем приложение на Windows/Linux когда все окна закрыты app.on('window-all-closed', () => { if (process.platform !== 'darwin') app.quit(); }); // Для MacOS: создаем окно если кликнули на иконку в доке app.on('activate', () => { if (BrowserWindow.getAllWindows().length === 0) createWindow(); });
